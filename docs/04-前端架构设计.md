## 三、路由配置

### 3.1 路由定义（App.tsx）

```typescript
import { Suspense, lazy } from 'react'
import { Routes, Route } from 'react-router-dom'

// 路由级代码分割
const HomePage = lazy(() => import('./pages/HomePage'))
const MovieDetailPage = lazy(() => import('./pages/MovieDetailPage'))
const SeriesDetailPage = lazy(() => import('./pages/SeriesDetailPage'))
const PlayerPage = lazy(() => import('./pages/PlayerPage'))

function App() {
  return (
    <Suspense fallback={<LoadingSpinner />}>
      <Routes>
        <Route path="/" element={<HomePage ... />} />
        <Route path="/movie/:id" element={<MovieDetailPage />} />
        <Route path="/series/:id" element={<SeriesDetailPage />} />
        <Route path="/player/:type/:id/:episodeId?" element={<PlayerPage />} />
      </Routes>
    </Suspense>
  )
}
```

### 3.2 路由说明

| 路径 | 组件 | 说明 |
|------|------|------|
| `/` | HomePage | 首页，展示媒体库、分类、搜索 |
| `/movie/:id` | MovieDetailPage | 电影详情，内嵌播放器 |
| `/series/:id` | SeriesDetailPage | 剧集详情，展示季/集列表 |
| `/player/:type/:id/:episodeId?` | PlayerPage | 独立播放器页面 |

### 4.1 HomePage（首页）

**文件**: `client/src/pages/HomePage.tsx`

**功能**:
- 展示首页Hero轮播、最新添加、媒体库
- 支持三种视图模式：home（首页）、movies（电影库）、series（连续剧）
- 实现搜索过滤（支持中文名、拼音首字母、英文名）
- 分类筛选和评分排序
- 分页功能（每页24条）

**状态管理**:
```typescript
const [data, setData] = useState<HomePageData | null>(null)
const [recentlyWatched, setRecentlyWatched] = useState<WatchHistoryItem[]>([])
const [allMovies, setAllMovies] = useState<Movie[]>([])

const [selectedGenre, setSelectedGenre] = useState('全部')

```

**搜索逻辑**:
```typescript
const filteredItems = useMemo(() => {
  if (searchQuery.trim()) {
    const query = searchQuery.toLowerCase().trim()
    items = items.filter((item: MediaCardData) => {
      const titleCn = (item.title_cn || '').toLowerCase()
      const titleEn = (item.title_en || '').toLowerCase()
      const searchKey = (item.search_key || '').toLowerCase()
      return titleCn.includes(query) || titleEn.includes(query) || searchKey.includes(query)
    })
  }
  return items
}, [allItems, selectedGenre, searchQuery])
```

## 八、类型定义

### 8.1 核心类型（shared/types.ts）

```typescript
// 媒体卡片数据（用于列表展示）
interface MediaCardData {
  id: number
  type: 'movie' | 'series'
  title_cn: string | null
  title_en: string | null
  search_key: string | null
  filename_rating: number | null
  filename_genre: string | null
  imdb_rating: number | null
  douban_rating: number | null
  local_poster: string | null
  release_date: string | null
  runtime: number | null
  episode_count?: number
  season_label?: string
}

// 完整电影数据
interface Movie extends MediaCardData {
  file_path: string
  file_size: number | null
  file_mtime: string | null
  ext: string
  tmdb_id: number | null
  overview: string | null
  director: string | null
  cast: string | null
  poster_path: string | null
  genres: string | null
  tmdb_rating: number | null
  vote_count: number | null
  metadata_status: 'pending' | 'fetched' | 'failed'
  metadata_updated: string | null
}

// 完整剧集数据
interface Series {
  id: number
  dir_path: string
  title_cn: string | null
  title_en: string | null
  search_key: string | null
  season_count: number
  episode_count: number
  season_label: string | null
  local_poster: string | null
  // ... 更多字段
}

// 单集数据
interface Episode {
  id: number
  series_id: number
  season_number: number
  episode_number: number
  episode_title: string | null
  watched: number
  progress: number
}

// 观看历史
interface WatchHistoryItem {
  id: number
  media_type: 'movie' | 'episode'
  media_id: number
  series_id: number | null
  progress: number
  title_cn?: string | null
  local_poster?: string | null
  episode_number?: number
  season_number?: number
}

// 首页数据
interface HomePageData {
  hero: MediaCardData | null
  recentlyWatched: WatchHistoryItem[]
  latestAdditions: MediaCardData[]
  library: PaginatedMedia
}
```
